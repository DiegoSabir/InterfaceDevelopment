#############################################################  PRODUCTOS  #################################################################

				Connection.save_product()  
Products.add_product() --> 
				Connection.show_products() --> Connection.select_products() --> Products.load_products_table


--Proceso para rellenar de datos tras pulsar un elemento de la tabla

Products.laod_products()

Products.modify_product() -->   
				Connection.show_products() --> Connection.select_products() --> Products.load_products_table(register)

Products.remove_product() -->	Products.clear_products()

				Connection.show_products() --> Connection.select_products() --> Products.load_products_table(register)
